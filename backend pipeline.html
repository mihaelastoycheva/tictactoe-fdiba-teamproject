name: Backend Build and Test


on:

  push:

    branches:

      - main

  pull_request:

    branches:

      - main


jobs:

  build:

    runs-on: ubuntu-latest


    steps:

    - name: Checkout code

      uses: actions/checkout@v2


    - name: Set up Node.js

      uses: actions/setup-node@v2

      with:

        node-version: '14'


    - name: Install dependencies

      run: npm install


    - name: Run backend tests

      run: npm test
